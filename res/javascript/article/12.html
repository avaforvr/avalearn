原生js添加移除class

添加class

/**
 * @description 为元素添加一个class
 * @param target DOM元素
 * @param className 类名
 */
function addClass(target, className) {
    if (!target) {
        return;
    }
    var classArr = target.className.split(' ');
    for (var i = 0; i < classArr.length; i++) {
        if (classArr[i] == className) {
            return;
        }
    }
    classArr.push(className);
    target.className = classArr.join(' ');
};

移除class

/**
 * @description 移除元素的某个class
 * @param target DOM元素
 * @param className 类名
 */
function removeClass(target, className) {
    if (!target) {
        return;
    }
    var classArr = target.className.split(' ');
    for (var i = 0; i < classArr.length; i++) {
        if (classArr[i] == className) {
            classArr.splice(i, 1);
        }
    }
    target.className = classArr.join(' ');
};
